import numpy as np
import matplotlib.pyplot as plt
# 数据
t = np.array([0.5, 1, 1.5, 2, 3, 4, 6, 9, 12, 18, 24])
C = np.array([19.96, 27.49, 29.33, 28.64, 24.82, 20.62, 13.90, 7.63, 4.19, 1.26, 0.38])
# 计算k
p1 = np.polyfit(t[-3:], np.log(C[-3:]), 1)
Cr = np.exp(p1[0] * t[:3] + p1[1]) - C[:3]
print(f'k = {-p1[0]:.2f}')
# 计算ka
p2 = np.polyfit(t[:3], np.log(Cr), 1)
print(f'ka = {-p2[0]:.2f}')
# 绘图
plt.plot(t, np.log(C), "*", label="Data")
plt.plot(t, np.polyval(p1, t), label="尾段直线")
plt.plot(t[:4], np.polyval(p2, t[:4]), label="残数线")
plt.text(10, 2.5, f"y = {p1[0]:.2f}x + {p1[1]:.2f}", fontsize=12)
plt.text(2.5, 1, f"y = {p2[0]:.2f}x + {p2[1]:.2f}", fontsize=12)
plt.xlabel("t")
plt.ylabel("ln(c)")
plt.legend()
plt.show()

3. (30分)

答案1：使用血药浓度数据计算，残数法

\begin{enumerate}
  \item[(1)] 对消除相的$\ln (C)$对$t$进行回归处理，得出尾段直线的回归方程，根据斜率求出$k$与消除半衰期。
  \item[(2)] 将吸收相各时间代入尾段直线的回归方程便能求出外推浓度
  \item[(3)] 外推浓度$-$实测浓度$=$残数浓度$C_\mathrm{r}$
  \item[(4)] 以残数浓度对数$\ln C_\mathrm{r}$对$t$作线性回归得残数线，从残数线的斜率求出$k_\mathrm{a}$
\end{enumerate}

答案2：使用血药浓度数据计算，W-N法
\begin{enumerate}
    \item[(1)] 选用尾段消除相的实测浓度-时间数据，以浓度对数对$t$作图，采用线性回归法求得尾段直线回归方程 ， 根据直线斜率求得消除速率常数$k$。
    \item[(2)] 根据实测浓度-时间数据，用梯形法分别求得从0到$t$时间点的$AUC_{0\to t_i}$以及$C{_t{_i}}+k\int_{0}^{t_i}C\mathrm{d}t=C_t{_i}+kAUC_{0\to t_i}$
    \item[(3)] 根据最后一点实测血药浓度$C_\mathrm{n}$与$k$值，求得$k\int_{0}^{\infty}C\mathrm{d}t=k\cdot AUC_{0\to t_n}+C_n$
    \item[(4)] 计算$t{_i}$时间点的药物吸收分数
    \item[(5)] 计算$t{_i}$时间点的待吸收分数，以其对数对$t$作图，求得直线回归方程，根据直线的斜率求得吸收速率常数$k_\mathrm{a}$
  \end{enumerate}


  答案3：使用尿药量数据计算


  \begin{enumerate}
    \item[(1)] 以$\frac{\Delta X_{\mathrm{u}}}{\Delta t}$的对数对$t_\mathrm{c}$作图，从直线的斜率可求出消除速率常数$k$值。
    \item[(2)] 计算尿药排泄总量：$X_\mathrm{u}^\infty = (X_\mathrm{u})_0^t + \frac{(\Delta X_\mathrm{u}/\Delta t)_t}{k} $
    \item[(3)] 以$\ln{(X_\mathrm{u}^\infty - X_\mathrm{u})}$对$t$作图，由尾段直线的斜率可求出消除速率常数$k$。
    \item[(4)] 将吸收相时间点代入回归方程求得吸收相$X_\mathrm{u}^\infty - X_\mathrm{u}$的外推值，再减去吸收相$X_\mathrm{u}^\infty - X_\mathrm{u}$的实测值，得残数亏量值。
    \item[(5)] 以残数亏量值的对数$\ln (X_\mathrm{u}^\infty - X_\mathrm{u})_r$对$t$作线性回归得残数线，从残数线的斜率求出$k_\mathrm{a}$
  \end{enumerate}

{\bf 评分标准：}可使用任意方法，合理即可。
