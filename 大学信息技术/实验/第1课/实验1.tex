\begin{frame}
    \frametitle{网络实名制}
    \begin{center}
        \cbf{中华人民共和国网络安全法}

\small（2016年11月7日第十二届全国人民代表大会常务委员会第二十四次会议通过）

    \end{center}
    \cbf{第二十四条}\quad 网络运营者为用户办理网络接入、域名注册服务，办理固定电话、移动电话等入网手续，或者为用户提供信息发布、即时通讯等服务，在与用户签订协议或者确认提供服务时，\bf{应当要求用户提供真实身份信息。用户不提供真实身份信息的，网络运营者不得为其提供相关服务}。国家实施网络可信身份战略，支持研究开发安全、方便的电子身份认证技术，推动不同电子身份认证之间的互认。
\end{frame}

\begin{frame}
    \frametitle{网络实名制}
    \begin{center}
        \cbf{互联网用户账号信息管理规定}

\small（2022年6月9日国家互联网信息办公室2022年第11次室务会议审议通过）

    \end{center}
    \cbf{第十二条}\quad 互联网信息服务提供者应当在互联网用户账号信息页面展示合理范围内的互联网用户账号的互联网协议\bf{（IP）地址归属地}信息，便于公众为公共利益实施监督。

    \cbf{第十三条}\quad 互联网信息服务提供者应当在互联网用户公众账号信息页面，展示公众账号的运营主体、注册运营地址、内容生产类别、统一社会信用代码、有效联系方式、互联网协议\bf{（IP）地址归属地}等信息。
\end{frame}
